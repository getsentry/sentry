import {Container, Flex} from '@sentry/scraps/layout';
import {IndeterminateLoader} from '@sentry/scraps/loader';

import * as Storybook from 'sentry/stories';

export const documentation = import('!!type-loader!@sentry/scraps/loader');

The `IndeterminateLoader` renders a repeating squiggle pattern with an animated multi-color gradient to indicate indeterminate loading progress. It fills 100% of its parent's width.

## Usage

<Storybook.Demo>
  <IndeterminateLoader />
</Storybook.Demo>

```jsx
<IndeterminateLoader />
```

## Contained Width

Constrain the loader's width by wrapping it in a sized container.

<Storybook.Demo>
  <Flex direction="column" gap="lg">
    <Container style={{width: 120}}>
      <IndeterminateLoader />
    </Container>
    <Container style={{width: 240}}>
      <IndeterminateLoader />
    </Container>
    <IndeterminateLoader />
  </Flex>
</Storybook.Demo>

```jsx
<Container style={{width: 120}}>
  <IndeterminateLoader />
</Container>

<Container style={{width: 240}}>
  <IndeterminateLoader />
</Container>

<IndeterminateLoader />
```

## Accessibility

The component renders with `role="progressbar"` and a default `aria-label` of `"Loading"`. Override the label to provide more specific context:

```jsx
<IndeterminateLoader aria-label="Saving changes" />
```
