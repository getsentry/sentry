import * as Storybook from 'sentry/stories';

import {Stack} from './stack.tsx';
import {Flex} from './flex.tsx';

export function Box(props) {
  return (
    <Flex
      background="primary"
      border="muted"
      style={{width: '64px', height: '64px'}}
      align="center"
      justify="center"
      {...props}
    />
  );
}
