import {Fragment, useState} from 'react';

import {Text} from '@sentry/scraps/text';

import * as Storybook from 'sentry/stories';

export const documentation = import('!!type-loader!@sentry/scraps/slider');

`<Slider>` allows users to select a numeric value by dragging a thumb along a track. It shows the current value above the thumb, with min/max labels that fade in during interaction.

```jsx
const [value, setValue] = useState(50);

<Slider value={value} onChange={setValue} min={0} max={100} />;
```

## Basic Usage

The simplest slider with default range (0-100):

export function BasicDemo() {
  const [value, setValue] = useState(50);
  return (
    <Stack gap="md" flexGrow={1} padding="3xl">
      <Text size="lg" tabular>
        Value: {value}
      </Text>
      <Slider value={value} onChange={setValue} aria-label="Basic slider" />
    </Stack>
  );
}

## Tick Marks

Tick marks are purely visual indicators — they are independent from `step`. The `ticks` prop accepts an object with `count` for evenly spaced ticks, `interval` for fixed spacing, or `values` for explicit positions.

export function TicksDemo() {
  const [value1, setValue1] = useState(50);
  const [value2, setValue2] = useState(50);
  const [value3, setValue3] = useState(50);
  return (
    <Stack gap="xl" flexGrow={1} padding="3xl">
      <Stack gap="md" flexGrow={1}>
        <Text size="lg" tabular>
          11 tick marks (value: {value1})
        </Text>
        <Slider
          value={value1}
          onChange={setValue1}
          ticks={{count: 11}}
          aria-label="Slider with tick count"
        />
      </Stack>
      <Stack gap="md" flexGrow={1}>
        <Text size="lg" tabular>
          Ticks every 25 (value: {value2})
        </Text>
        <Slider
          value={value2}
          onChange={setValue2}
          ticks={{interval: 25}}
          aria-label="Slider with tick interval"
        />
      </Stack>
      <Stack gap="md" flexGrow={1}>
        <Text size="lg" tabular>
          Ticks at specific values (value: {value3})
        </Text>
        <Slider
          value={value3}
          onChange={setValue3}
          ticks={{values: [0, 10, 25, 50, 75, 100]}}
          aria-label="Slider with tick values"
        />
      </Stack>
    </Stack>
  );
}

```jsx
// 11 evenly spaced tick marks

// Tick marks every 25 units

```

Note: ticks and step are decoupled. A slider with `step={1}` (smooth dragging) can display `ticks={{count: 5}}` as visual markers.

## Label Formatting

Use the `formatLabel` callback to customize how values are displayed in the value label, edge labels, and tick labels:

export function FormatDemo() {
  const [value1, setValue1] = useState(50);
  const [value2, setValue2] = useState(500);
  return (
    <Stack gap="xl" flexGrow={1} padding="3xl">
      <Stack gap="md" flexGrow={1}>
        <Text size="lg" tabular>
          Percentage: {value1}%
        </Text>
        <Slider
          value={value1}
          onChange={setValue1}
          formatLabel={v => `${v}%`}
          aria-label="Percentage slider"
        />
      </Stack>
      <Stack gap="md" flexGrow={1}>
        <Text size="lg" tabular>
          Currency: ${value2}
        </Text>
        <Slider
          value={value2}
          onChange={setValue2}
          min={0}
          max={1000}
          step={50}
          formatLabel={v => `$${v}`}
          aria-label="Currency slider"
        />
      </Stack>
    </Stack>
  );
}

## Controlled vs Uncontrolled

Sliders can be controlled or uncontrolled:

```jsx
// Controlled (recommended)
const [value, setValue] = useState(50);
<Slider value={value} onChange={setValue} />

// Uncontrolled
<Slider defaultValue={50} />
```

## Keyboard Controls

`<Slider>` provides full keyboard support:

- **Arrow Left/Down**: Decrease value by step
- **Arrow Right/Up**: Increase value by step
- **Home**: Jump to minimum value
- **End**: Jump to maximum value
- **Page Up/Down**: Larger increments (10× step)

## Accessibility

`<Slider>` follows the [WAI-ARIA Slider pattern](https://www.w3.org/WAI/ARIA/apg/patterns/slider/) and meets WCAG 2.2 AA standards:

- [2.1.1 Keyboard](https://www.w3.org/TR/WCAG22/#keyboard): Full keyboard control
- [2.4.7 Focus Visible](https://www.w3.org/TR/WCAG22/#focus-visible): Clear focus indicators

The component automatically includes:

- `role="slider"`
- `aria-valuemin`, `aria-valuemax`, `aria-valuenow` attributes
- `aria-valuetext` for formatted values
- Keyboard navigation

### Developer Responsibilities

**Labels**

- Every slider must have an associated label
- Use `<label>`, `aria-label`, or `aria-labelledby`

```jsx
// Good: Visible label
<label>
  Volume
  <Slider value={volume} onChange={setVolume} />
</label>

// Good: aria-label
<Slider value={volume} onChange={setVolume} aria-label="Volume level" />
```

**Value Communication**

- Use `formatLabel` to provide meaningful value displays
- The slider's built-in value label, edge labels, and tick labels all use `formatLabel`

**Keyboard Interaction**

- Test that all keyboard shortcuts work correctly
- Ensure focus indicators are visible
- Verify that Home/End keys reach min/max values

For more information, see the [WAI-ARIA Slider practices](https://www.w3.org/WAI/ARIA/apg/patterns/slider/).
