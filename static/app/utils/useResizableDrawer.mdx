## useResizableDrawer

Use this hook to create a resizable drawer.

### Return

- `resizing`: A boolean that indicates if the drawer is resizing.
- `resizeHandleProps`: The props to pass to the resize handle.
- `resizedElementProps`: The props to pass to the element you wish to resize.

```tsx
function App() {
  const { resizing, resizeHandleProps, resizedElementProps } = useResizableDrawer({
    direction: 'right',
  });

  return (
    <div>
      <div {...resizedElementProps}>
        <div {...resizeHandleProps}>Resize</div>
      </div>
    </div>
  );
}
```

## Resizing Width

export function Demo() {
  const { resizing, resizeHandleProps, resizedElementProps } = useResizableDrawer({
    direction: 'right',
    initialSize: {width: 200},
    max: {width: 400},
    min: {width: 100},
  });

  return (
    <div>
      <div
        {...resizedElementProps}
        style={{ padding: '10px', border: '1px solid #ccc' }}
      >
        <div {...resizeHandleProps}>Resize to max 400px and min 100px</div>
      </div>
    </div>
  );
}

## Resizing Height

export function DemoHeight() {
  const [size, setSize] = useState(200);
  const { resizeHandleProps } = useResizableDrawer({
    direction: 'down',
    initialSize: {width: 200, height: 200},
  });

  return (
    <div>
      <div
        {...resizeHandleProps}
        style={{ padding: '10px', border: '1px solid #ccc', height: size }}

      </div>
