{% trans "load the packages in the namespace:" %}

library(logging)
library(logging.handlers)

{% trans "associate a sentry handler to a logger:" %}

rl <- getLogger() # for example the root logger
rl$addHandler(sentryHandler, dsn=DSN)

{% trans "logging records going through the logger will be handled:" %}

sol <- getLogger('some.other.logger')
sol$error(MESSAGE)
sol$warn(MESSAGE)
sol$info(MESSAGE)
sol$debug(MESSAGE)
